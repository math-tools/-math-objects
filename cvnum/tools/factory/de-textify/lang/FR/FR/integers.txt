sign::
    + = plus
    - = moins

general::
    sep = .
    big = milliard... -> ... de milliards

group::
    3  = [d :if: d > 1]
         mille
         [r]
    6 => million{s}
    9 => "de :if: d(0..5) = 0"
         milliard{s}
